<!-- The state contract this plan establishes. Phase 2 will extend appState.screen with 'intake', 'loading', 'result', 'error', and Phase 2/3 will add fields like intake answers and parsed cheat sheet. The shape below MUST hold. -->

appState shape:
  {
    screen: 'key-entry' | 'ready',   // current screen
    apiKey: string | null            // raw key, in memory only, never put in innerHTML
  }

setState(partial):
  Object.assign(appState, partial)
  render()

render():
  reads appState.screen
  toggles the hidden attribute on the pre-rendered <section id="keyEntry"> and <section id="ready">
  when screen === 'ready', writes maskKey(appState.apiKey) to #maskedKey via textContent

localStorage key:
  'migrationsamt.anthropicKey'   // namespaced; getItem / setItem / removeItem only

maskKey(key):
  if !key or key.length < 12, return '(hidden)'
  return key.slice(0, 7) + '...' + key.slice(-4)
</interfaces>
